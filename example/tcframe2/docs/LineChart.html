TcFrame v2.0 API Docs
示例
  var LineChart=new TcFrame.LineChart({
    x:0,y:0,width:500,height:300,
    minX:0,valueOfPixX:1,minY:0,valueOfPixY:5,
    title:"部门利润图", comments:["X:时间（月）","Y:利润（万元）"]
    backgroundMeshShow:true,autoFillAxisX:true,autoFillAxisY:true
  });
  LineChart.dataProvider([
    {
      //第一条折线的说明文字，即含义
      label:"总体",
      //第一条折线的颜色
      color:"rgba(51,153,255,0.8)",
      //第一条折线的线宽度
      lineWidth:2,
      //第一条折线上方折线的点阵
      data:[
        [0,1000],[100,1300],[200,1200],
        [300,1000],[400,1150],[500,890],
        [600,800],[700,920]
      ]
    },{
      label:"部门3",color:"rgba(255,0,0,0.5)",lineWidth:2,
      data:[[0,500],[100,300],[200,150],[300,450],[400,480],[500,620],[600,1000],[700,900]]
    },{
      label:"部门1",color:"rgba(255,255,0,0.5)",lineWidth:2,
      data:[[0,450],[100,400],[200,350],[300,300],[400,350],[500,370],[600,400],[700,300]]
    },{
      label:"部门2",color:"rgba(0,0,50,0.5)",lineWidth:2,
      data:[[0,350],[100,300],[200,250],[300,200],[400,250],[500,270],[600,300],[700,200]]
    }
  ]);
