TcFrame v2.0 API Docs
示例
function okFunc(){window.alert('确定');}
function cancelFunc(){window.alert('取消');}
function returnfunc(a){window.alert(a);}
TcFrame.Alert.show({
   title:"友情提示",
   text:"普通信息提示",
   titleAlign:"center",
   labelAlign:"left"
});
TcFrame.Alert.confirm({
   title:"友情提示",
   text:"确定取消选择",
   titleAlign:"center",
   labelAlign:"left",
   okFunc:okFunc,
   cancelFunc:cancelFunc
});
TcFrame.Alert.prompt({
   title:"友情提示",
   text:"输入一个网址",
   defaultValue:"http://",
   titleAlign:"center",
   labelAlign:"left",
   func:returnfunc
});
