示例
  var c=new TcFrame.RingPaper({x:50,y:50,width:800,height:350,title:"饼型数据图"});
  c.DrawRing({
    outsideDiameter:100,internalDiameter:20,
    startAngle:0,endAngle:Math.PI,
    stroke:false,
    fillStyle:"#FF0000",text:"数据1"
  });
  c.DrawRing({
    outsideDiameter:120,internalDiameter:20,
    startAngle:Math.PI,endAngle:Math.PI*1.3,
    stroke:false,
    fillStyle:"#339966",text:"数据2"
  });
  c.DrawRing({
    outsideDiameter:90,internalDiameter:20,
    startAngle:Math.PI,endAngle:Math.PI*1.3,
    stroke:false,
    fillStyle:"#3399ff",text:"数据3"
  });
  c.DrawRing({
    outsideDiameter:100,internalDiameter:20,
    startAngle:Math.PI*1.3,endAngle:Math.PI*1.8,
    stroke:false,
    fillStyle:"#333",text:"数据4"
  });
  c.DrawRing({
    outsideDiameter:100,internalDiameter:20,
    startAngle:Math.PI*1.8,endAngle:Math.PI*2,
    stroke:false,
    fillStyle:"#65cc65",text:"数据6"
  });
  c.DrawRing({
    outsideDiameter:90,internalDiameter:30,
    startAngle:-Math.PI*0.1,endAngle:Math.PI*0.8,
    stroke:false,
    fillStyle:"#995599",text:"数据7"
  });
