TcFrame v2.0 API Docs
示例
var vGroup=new TcFrame.Group({display:"vertical",left:5,top:5,bottom:5,width:80});
vGroup.setStyle('overflow','auto');
application.add(vGroup);
for(var n=0;n<50;n++){
   var canvas=new TcFrame.Canvas({width:50,height:50});
   canvas.content.innerHTML=n;
   canvas.setStyle(
      'backgroundColor',
      "rgba("+parseInt(Math.random()*255)+","+parseInt(Math.random()*255)+","+parseInt(Math.random()*255)+",1)")
   );
   vGroup.add(canvas);
}
