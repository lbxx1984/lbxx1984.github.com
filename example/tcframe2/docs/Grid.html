TcFrame v2.0 API Docs
示例
var grid=new TcFrame.Grid({left:670,right:5,top:35,bottom:40});
var gridDate={
  column:[
   {width:80,label:'A'},{width:100,label:'B'},{width:40,label:'C'},
   {width:80,label:'D'},{width:80,label:'E'},{width:80,label:'F'},
   {width:80,label:'G'},{width:80,label:'H'},{width:80,label:'I'},
   {width:80,label:'J'},{width:80,label:'K'}
  ],//表格列的信息，包括列头标号和列的宽度
  data:[]//二维数组，数组的一行代表表格中的一行
}
for(var n=0;n<30;n++){//给表格添加30行数据
  var it=[];
  for(var m=0;m<11;m++){it.push(parseInt(Math.random()*100));}//每行添加11个数据
  gridDate.data.push(it);
}
grid.dataProvider(gridDate);
